// ── Helpers ──

function timeAgo(ts) {

    const d = Math.floor(Date.now()/1000) - ts;

    if (d < 60)    return d + 's';

    if (d < 3600)  return Math.floor(d/60) + 'm';

    if (d < 86400) return Math.floor(d/3600) + 'h';

    if (d < 604800) return Math.floor(d/86400) + 'd';

    return new Date(ts*1000).toLocaleDateString();

}


function fmtN(n) {

    if (!n) return '0';

    if (n >= 1e6) return (n/1e6).toFixed(1)+'M';

    if (n >= 1e3) return (n/1e3).toFixed(1)+'K';

    return String(n);

}


function esc(s) {

    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

}


const IMG_RE = /(https?:\/\/\S+\.(?:jpe?g|png|gif|webp|svg|avif)(?:\?\S*)?)/gi;

const VID_RE = /(https?:\/\/\S+\.(?:mp4|webm|mov)(?:\?\S*)?)/gi;


function extractMedia(text) {

    const imgs = [...text.matchAll(IMG_RE)].map(m => m[1]);

    const vids = [...text.matchAll(VID_RE)].map(m => m[1]);

    return { imgs, vids };

}


function renderContent(raw) {

    let text = raw.replace(IMG_RE, '').replace(VID_RE, '').trim();

    text = esc(text);


    // URLs → links

    text = text.replace(/(https?:\/\/[^\s<]+)/g, '<a href="$1" target="_blank" rel="noopener noreferrer">$1</a>');


    // Hashtags

    text = text.replace(/#(\w{1,60})/g, '<span class="ht">#$1</span>');


    // nostr: mentions

    text = text.replace(/nostr:(npub|nprofile|note|nevent|naddr)1[a-z0-9]{10,}/gi, (match) => {

        const short = match.slice(0,20) + '…';

        return `<span class="mention">${short}</span>`;

    });


    return text;

}

        const name = p.display_name || p.displayName || p.name || note.pubkey.slice(0,10)+'…';

}


// ── Primal Cache API ──

function fetchFeed() {

    // Reset

    profiles = {}; notes = []; noteStats = {};

    document.getElementById('feed').innerHTML = `

        <div class="state-message">

            <div class="spinner"></div>

            <span>Loading ${activeFeed} feed…</span>

        </div>`;


    if (ws) { try { ws.close(); } catch(_){} }

    clearTimeout(connectTimeout);


    ws = new WebSocket(PRIMAL_WS);


    // Timeout → fallback to raw relays

    connectTimeout = setTimeout(() => {

        if (ws.readyState !== WebSocket.OPEN) {

            ws.close();

            fetchFromRelays();

        }

    }, 6000);


    ws.onopen = () => {

        clearTimeout(connectTimeout);

        ws.send(JSON.stringify([

            "REQ",

            "wire",

            { "cache": [ FEED_METHODS[activeFeed], { "limit": 40 } ] }

        ]));

    };


    ws.onmessage = (e) => {

        let msg;

        try { msg = JSON.parse(e.data); } catch(_) { return; }


        if (msg[0] === 'EVENT' && msg[2]) {

            const ev = msg[2];

            switch (ev.kind) {

                case 0: // profile

                    try { profiles[ev.pubkey] = JSON.parse(ev.content); } catch(_){}

                    break;

                case 1:  // text note

                case 30023: // long-form

                    notes.push(ev);

                    break;

                default:

                    // Primal stats kinds (10000100+)

                    if (ev.kind >= 10000100 && ev.kind <= 10000200) {

                        try {

                            const d = JSON.parse(ev.content);

                            if (d.event_id) noteStats[d.event_id] = d;

                        } catch(_){}

                    }

            }

        }


        if (msg[0] === 'EOSE') {

            renderFeed();

            try { ws.close(); } catch(_){}

        }

    };


    ws.onerror = () => {

        clearTimeout(connectTimeout);

        fetchFromRelays();

    };

}


// ── Fallback: Direct Relay Fetch ──

function fetchFromRelays() {

    profiles = {}; notes = []; noteStats = {};

    document.getElementById('feed').innerHTML = `

        <div class="state-message">

            <div class="spinner"></div>

            <span>Trying direct relay connection…</span>

        </div>`;


    let connected = false;


    for (const url of FALLBACK_RELAYS) {

        const r = new WebSocket(url);

        r.onopen = () => {

            if (connected) { r.close(); return; }

            connected = true;


            // Fetch recent notes

            r.send(JSON.stringify(["REQ","fb",{"kinds":[1],"limit":40}]));


            const pubkeys = new Set();
